  become_method: runas
  tasks:

      win_file:

        path: C:\Users\ansible\Documents\jaden.txt
        state: file

    - name: Write text in file
      ansible.windows.win_copy:
        dest: C:\Users\ansible\Documents\jaden.txt
        content: |
          This is a test
          i can write on multiple different lines.
